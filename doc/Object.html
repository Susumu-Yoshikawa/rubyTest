def fizz_buzz(n)
  if n % 15 == 0
    'FizzBuzz'
  elsif n % 3 == 0
    'fizz'
  elsif n % 5 ==0
    'buzz' 
  else
    n.to_s
  end
end

def to_hex(r,g,b)
    [r,g,b].inject('#') do |hex, n|
      hex + n.to_s(16).rjust(2, '0')
    end
end
